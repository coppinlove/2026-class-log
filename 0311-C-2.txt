// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int num;
    
    
    printf("숫자를 입력하세요:");
    scanf("%d", &num);
    
    
    if (num % 2 == 0) {
        printf("짝수");
        printf("\n2로 나눈 나머지가 0");
    }
    
    else {
        printf("홀수");
        printf("\n2로 나눈 나머지가 1");
    }


    return 0;
}

/*if = 조건문
else = 조건이 아닐 때
% = 나머지
/ = 몫
> = 초과
< = 미만
>= = 이상
<= = 이하
== = 같다
= = 값 지정
!= = 같지않다
if를 사용할때 if 뒤에 소괄호 안에 조건문 넣고 실행값을 쓸 때 사이에 ; 사용X
한 묶음으로 사용하기 위해서 조건문 뒤에 {} 넣어서 묶기*/
