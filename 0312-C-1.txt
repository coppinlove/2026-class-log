#include <stdio.h>

int main()
{
    int num;
    
    
    while(1)
    {
    
    
    printf("숫자를 입력하세요:");
    scanf("%d", &num);
    
    
    if (num == 10) {
        printf("정답입니다.");
        break;
    }
    
    
    else if (num > 10) {
        printf("\n너무 높아요.\n\n\n");
    }
    
    
    else{
        printf("\n너무 낮아요.\n\n\n");
    }


    }
    return 0;
}
